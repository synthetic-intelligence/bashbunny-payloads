#!/bin/bash

# Initialize
LED SETUP 

# Configuration
SOURCE="/dev/nandf"
SDHC="/dev/mmcblk0p1"
SDHC_MOUNT="/root/udisk"
DEST="${SDHC_MOUNT}/backups"
PREFIX="nandf-backup"
COMPRESS_LEVEL="1"
OUTPUT_PREFIX="${DEST}/${PREFIX}-"
OUTPUT_SUFFIX="1"

# Backup

LED GREEN
if [ -b ${SDHC} ]; then
  mount -o sync ${SDHC} ${SDHC_MOUNT}
else
  LED FAIL
  exit 1
fi

while [[ -e "${OUTPUT_PREFIX}${OUTPUT_SUFFIX}.tar.gz" ]]; do
    ((${OUTPUT_SUFFIX}++))
done

dd if="$SOURCE" bs=4M | gzip -"${COMPRESS_LEVEL}" > "${OUTPUT_PREFIX} ${OUTPUT_SUFFIX}" && \
LED FINISH
